# 프로토콜(Protocols)
```swift

//프로토콜은 영어로 규약이 나 협약이라고도 부른다.
//어떤 자격증 같은거라고 생각하면 될듯

protocol 햄버거제조 {
    func 고기굽기()
}


struct 버거킹: 햄버거제조 {
    func 고기굽기() {
        print("고기를 굽습니다~")
    }
}


class 맥더날드: 햄버거제조 {
    func 고기굽기() {
        print("고기를 조금만 굽습니다~")
    }
}

//프로토콜은 클래스의 상속의 단점을 보완 할수 있다 ======================

//프로토콜은 클래스의 상속의 단점을 보완 할수 있다 ======================


protocol 고기패티제조 {
    func 고기굽기()
}

class 햄버거가게 {
    var 드라이빙스루 = true
    
    func 자동차주문() {
        if 드라이빙스루 {
            print("자동차 주문이 가능합니다")
        }
    }
}


class 버거킹: 햄버거가게,고기패티제조 {   //자격증을 채택 이렇게 채택한곳에서 구체적인 구현을해야함
    func 고기굽기() {
        print("고기를 굽습니다")
    }
    
    func 채소손질() {
        print("채소를 손질합니다")
    }
    
}


class 맘스터치: 햄버거가게 {
    func 패티튀기기() {
        print("치키패티를 튀깁니다")
    }
}


struct 맥더날드: 고기패티제조 {  // 구조체에서 채택도 가능
    func 고기굽기() {
        print("고기를 굽습니다")
    }
}



struct 햄버거프렌차이즈 {
    func 고기패티햄버거(고기패티: 고기패티제조) {
        고기패티.고기굽기()
    }
}

let 버거킹1 = 버거킹()

let 맘스터치1 = 맘스터치()

let 맥더날드1 = 맥더날드()

let 햄버거프렌차이즈1 = 햄버거프렌차이즈()


햄버거프렌차이즈1.고기패티햄버거(고기패티: 버거킹1)
햄버거프렌차이즈1.고기패티햄버거(고기패티: 맥더날드1)
//햄버거프렌차이즈1.고기패티햄버거(고기패티: 맘스터치1) //고기패티제조라는 프로토콜이 채택되지않아서 안됨 자격증이없는걸

```
