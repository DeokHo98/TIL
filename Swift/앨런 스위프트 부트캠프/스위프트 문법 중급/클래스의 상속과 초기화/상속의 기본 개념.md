//클래스의 상속
//상속이라는 기능은 유일하고 클래스에서만 가능한 기능이다. 구조체는 x
//상손은 쉽겜ㄹ해, 본질적으로 성격이 비슷한 타입을 새로만들어
//데이터(저장속성)을 축하거나 기능(메서드)를 변형시켜 사용하는것이다.

//상속의 기본문법 ==========================================================================================

class Aclass {
    var name = "name"
}

class Bclasss: Aclass { //이렇게 : 슈퍼클래스 이름을 붙힌다.
    var id = 0
}

let b = Bclasss()
b.id
b.name


//b클래스는 a클래스를 상속해서 구현
//서브클래스는 슈퍼클래스로 부터 멤버를 상속함

//클랙스의 상속 금지의 키워드 final ======================================================================
//클래스 이르앞에 final 키워드를 붙히면 상속금지를 시킬수있음
final class Cclass {
   var name = "name"
}

//class Dclasss: Cclass { 에러가나면서 상속할수 없게된다.
//    var id = 0
//}

   final var name = "name"
}
