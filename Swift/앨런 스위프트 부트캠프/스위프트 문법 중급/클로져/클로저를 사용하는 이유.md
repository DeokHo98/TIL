//클로저는 왜 이름이 필요없을까?
//클로저는 함수를 실행할때 전달하는 형태로 사용하기때문에 이름이 필요없다.





//클로저를 파라미터로 받는 함수를 정의

func 클로저함수(클로저: () -> ()) {
    print("함수 시작~")
    클로저()
}


// 함수를 실행할때 클로저를 파라미터로 할당하기

func 종료함수() {    //위에 파라미터의 타입만 같으면 된다.
    print("함수 종료~")
}

let 종료클로저 = { () -> () in
    print("함수 종료~~~~~~~~~~")
}


클로저함수(클로저: 종료함수)


클로저함수(클로저: 종료클로저)


//함수를 실행할때 클로저 형태로 전달하기 (이게 클로저를 사용하는 이유다)

클로저함수(클로저: {() -> () in
    print("프린트 종료한다 알았냐?")
})





//또다른 예시


func 클로저함수2(a: Int, b: Int, 클로저: (Int) -> () ) {
    let c = a * b
    클로저(c)
}

클로저함수2(a: 5, b: 5, 클로저: { (num) in
    print("곱한 값은 \(num) 입니다")
})


//실질적으로는 함수가 모두 실행된뒤에 클로저가 실행된다고 봐도 무방하다.
//클로저는 향후 개발자가 원하는데로 정의할수 있기때문에 활용도가 늘어난다.


let 프린트1 = {
    print("1")
}

let 프린트2 = {
    print("2")
}

func 멀티클로저함수(closure1: () -> (), closure2: () -> () ) {
    closure1()
    closure2()
}

멀티클로저함수(closure1: 프린트1, closure2: 프린트2)


멀티클로저함수(closure1: {
    print("1")
}, closure2: {
    print("2")
})
//이렇게 내가 내 입맛대로 정의할수도 있는것






func performClosure(closure: () -> ()) {
    print("시작")
    closure()
    print("끝")
}



performClosure(closure: {
    print("===============중간=================")
})
