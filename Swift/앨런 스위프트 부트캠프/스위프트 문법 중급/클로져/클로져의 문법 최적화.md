//클로저의 문법 최적화 =======================

/*
 문법 최적화(간소화)
 - 1) 문맥상에서 파라미터와 리턴밸류 타입 추론(Type Inference)
 - 2) 싱글 익스프레션인 경우(한줄), 리턴을 안 적어도 됨(Implicit Return)
 - 3) 아규먼트 이름을 축약(Shorthand Argements) ===> $0, $1
 - 4) 트레일링 클로저 문법: 함수의 마지막 전달 인자(아규먼트)로 클로저 전달되는 경우, 소괄호를 생략 가능
*/



// 트레일링 클로저 - 후행클로저 문법


func 클로저함수(closuer: () -> () ) {
    print("프린트 시작~")
    closuer()
}

//함수의 마지막 전달 인자로 클로저가 전달되는경우, 소괄호를 생략하는것이 가능

클로저함수(closuer: {
    print("프린트 끝..")
})

클로저함수(closuer: ) {     //소괄호를 앞으로 가져오는것도 가능
    print("프린트종료....")
}

클로저함수() {
    print("프린트 종료..") //아규먼트를 생략하는 것도 가능하다.
}

//소괄호를 아예 생략하는것도 가능
// 아래형태가 함수르 실행하고 마지막 아규먼트로 클로저를 전달했따는 형태이다.


클로저함수 {
    print("프린트 끝~~~")
}


//연습

func 클로저케이스함수(a: Int, b: Int, closure: (Int) -> ()) {
    let c = a * b
    closure(c)
}

클로저케이스함수(a: 5, b: 5) { num in //소괄호가 클로저 앞에서 닫히는 예시..
    print("곱한값은 \(num) 입니다")
}



//파라미터 및 생략 등의 간소화

func 클로저함수2(param: (String) -> Int) {
    param("가나다라마바사")
}


// 1. 타입 추론
 
클로저함수2(param: { (str: String) -> Int in
    return str.count
})

클로저함수2(param: { str in
    return str.count
})

// 2. 코드가 한줄인 경우 리턴을 안적어도 됨

클로저함수2(param: {str in
    str.count
})


//3. 아규먼트 이름을 축약
클로저함수2(param: { //아규먼트 이름을 안쓰고
    $0.count //달러표시를 쓰고 첫번째 파라미터라는 의미로 0 을써주면 된다.
})

//4. 트레일링 클로저

클로저함수2 { str in
    str.count
}

클로저함수2 {
    $0.count
}


//축약 형태로 활용

let 클로저타입1 = { (param) in
    return param % 2 == 0
}
클로저타입1(4)



let 클로저타입2 = { $0 % 2 == 0}
클로저타입2(3)



let 클로저타입3 = { (a: Int, b: Int) -> Int in
    return a * b
}
클로저타입3(5,5)



let 클로저타입4: (Int,Int) -> Int = { (a,b) in
    return a * b
}
클로저타입4(5,5)


let 클로저타입5: (Int,Int) -> Int =  {
    $0 * $1
}
클로저타입5(5,5)
