```swift

/*
- 어튜리뷰트 키워드
- @available, @objc, @escaping, @IBOutlet, @IBAction 등등


- Attribute(어트리뷰트) ===> 컴파일러에게 알려주는 특별한 신호이자, 추가적인 정보를 제공 (2가지 종류가 있음)


- 1) 선언에 대한 추가정보 제공
- 2) 타입에 대한 추가정보 제공


- 사용 방법
- @어트리뷰트이름          ======> (예시)  @available
- @어트리뷰트이름(아규먼트)  ======> (예시)  @available(iOS *)
*/


//사용 실제 예시 =====

@available(iOS 13.0, macOS 10.12, *)
class someType { //이클래스는 ios 13.0 macos 10.12 버전에서만 읽을수 있음
}





// 구현해도되고 안해도 되는 선택적인 멤버 선언하기 ===============================

/*
 - @objc는 스위프트로 작성한 코드를 오브젝티브C 코드에서도 사용할 수 있게 해주는 어트리뷰트
 - 프로토콜에서 요구사항 구현시, 반드시 강제하는 멤버가 아니라 선택적인 요구사항으로 구현할때 사용
 
 - 프로토콜 앞에는 "@objc"추가
 - 멤버 앞에는 "@objc optional"을 모두 추가
 */

//예시

@objc protocol 리모트 {
    @objc optional var 전원: Bool {get set}
    func 온()
    func 오프()
    @objc optional func 넷플릭스하기()
}

class 티비: 리모트 {
    var 전원 = false
    
    func 온() {}
    func 오프() {}
    
//    func 넷플릭스하기() { //구현안해도됨
//    }
}


/*
 - 선택적 멤버를 선언한 프로토콜 구현시
 - 오브젝티브-C에 해당하는 클래스 전용 프로토콜임 (구조체, 열거형 채용 불가) ⭐️
 - (오브젝티브C는 구조체와 열거형에서 프로토콜 채택을 지원하지 않음)
 - 그래서 사실상 @objc 키워드가 붙은 프로토콜은 사실상 클래스전용 프로토콜이다
 */



let 티비1 = 티비()
print(티비1.전원)


let 티비2: 리모트 = 티비()
print(티비2.전원) //@objc 키워드가 있기때문에 있을수도 없을수도 있기때문에 옵셔널이다
티비2.넷플릭스하기?()   // (선택적으로 선언했기 때문에, 함수가 없을 수도 있음 ===> 옵셔널체이닝 필요)




```
