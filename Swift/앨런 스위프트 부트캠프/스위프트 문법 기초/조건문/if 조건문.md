# 조건문(if문)
참 또는 거짓의 특정 조건에 따라 특정 코드만 실행 할 수있게 하는 문법
    
```swift
//if 문
var number = 10

if number <= 10 {
    print("10보다 작거나 같다.")
}
// if 뒤에는 변수 그리고 그뒤에는 조건, 조건의 자리에는 참 또는 거짓이 나오는 결과가 있어야한다. 그리고 중괄호를 열고 조건에 만족하면 실행하고 싶은 코드를 작성하면 된다.


//if - else 문

number = 20


if number <= 10 {
    print("10보다 작거나 같다.")
} else {
    print("10보다 크다.")
}
//위에랑 똑같지만 그 뒤에 else 가 추가된다. else는 "그 조건에 해당이 안되? 그럼 이 코드 실행하자" 와 같다. else를 쓰고 그뒤에는 만약 조건에 해당 되지 않는다면 실행하고 싶은 코드를 쓰면 된다.


//if - else if -else 문

number = 40

if number <= 10 {
    print("10보다 작거나 같다.")
} else if number >= 30 {
    print("30보다 크거나 같다.")
} else {
    print("10보다 크고 30보다 작거나 같다.")
}
//else if란 "그 조건에 해당이 안돼? 그럼 만약에 이조건은 어떄?" 라고 물어보는거와 같다. 쉽게 말해 첫번째 조건에 해당이 안된다면  두번째 조건을 내거는거다. else if는 2,3,4번 넘게 연속으로 중첩해서 사용이 가능하다.

//if - else if 문

number = 72

if number <= 10 {
    print("10보다 작거나 같다.")
} else if number >= 30 {
    print("30보다 크거나 같다.")
}

//바로 위에 if else if else문이랑 큰 차이는없는데, 그냥 쉽게 말해 조건을 두번까지 검사를 했는데도 조건에 부합하지않으면 그냥 넘어가자~ 그냥 여기 부분을 아예 뛰어넘어 버리자~ 하는 의미이다.
``` 
     
        
# if문에서 논리적인 구조 및 조건의 순서가 중요

1. 논리적인 오류 없도록 구성해야함

```swift
var num = 8

// 잘못된 예 ===============
if num >= 0 {
    print("0 또는 양수입니다.")
} else if num % 2 == 0 {
    print("양의 짝수입니다.")
} else if num % 2 == 1 {
    print("양의 홀수입니다.")
} else {
    print("음수입니다.")
}
//이 코드는 왜 잘못되었냐면 숫자 8 은 0또는 양수다. 그리고 또한 짝수이기도하다. 그런데 실행해보면 0또는양수입니다만 출력하게되는것이다. 그러므로 논리적으로 잘못된 코드다.
//참고로 % 2 == 0 은 2로나눴을때 나머지가 0이냐를 물어보는것이다. 당연히 나머지가 0이면 짝수 나머지가 1이면 홀수다.

// 올바르게 수정해 본다면? ================

if num >= 0 {
    print("0 또는 양수입니다.")
    if num % 2 == 0 {
        print("짝수입니다.")
    } else if num % 2 == 1 {
        print("홀수입니다.")
    }
    
} else {
    print("음수입니다.")
}
//어떻게 수정을 했냐면 첫번째 조건안에 조건을 또 중첩시킨것이다. 첫번째 조건 자체가 0또는 양수인지확인하고 또 짝수인지 홀수인지를 확인하는 조건인 것이다.
```
    
    
2. // 조건의 확인 순서를 잘 고려해야함 ==========
```swift

// 잘못된 예 ===============
var score = 100


if score >= 70 {
    print("70점이상입니다.")
} else if score >= 80 {
    print("80점이상입니다.")
} else if score >= 90 {
    print("90점이상입니다.")
} else {
    print("70점미만입니다.")
}
// 여기도보면 점수가 100점인데 첫번째 조건이 70점이상이면 100은 80점이상이기도하고 90점이상이기도 한데. 논리적을 또한 맞지가 않다.

// 올바르게 수정해 본다면? ================

if score >= 90 {                // 범위가 작은 조건이 먼저 와야함
    print("90점이상입니다.")
} else if score >= 80 {
    print("80점이상입니다.")
} else if score >= 70 {
    print("70점이상입니다.")
} else {
    print("70점미만입니다.")
}

//이렇게 범위가 작은조건의 순서대로 작성하는것이 중요하다. 90점 이상의 범위가 제일 적기 때문이다.
```

# 조건을 &&(and), 또는 ||(or)로 연결하는 것도 가능

```swift
var email = "a@gmail.com"
var password = "1234"



if email == "a@gmail.com" && password == "1234" {
    print("메인페이지로 이동")
}
//and 연산자를 사용한 것 인데. and연산자는 모두 같은지를 물어보는 것이다. 
//email이 저 값과 같고 패스워드도 저 값과 같다면 둘다 같다면 저 메시지를 프린트하는것이다.

if email != "a@gmail.com" || password != "1234" {
    print("경고메세지 띄우기")
    print("이메일주소 또는 패스워드가 올바르지 않습니다.")
}
//or 연산자를 사용한것
//이메일이나 패스워드 둘중하나가 저 값이랑 다르다면? 둘중 하나라도 다르다면 저 메시지를 프린트하는것이다.
```

