# 함수의 리턴값 유무
```swift
// 리턴값이 없는 함수
func sayHello() {
    print("안녕하세요.") //그냥 프린트동작만하고 함수의 결과값은 void 다. 아무것도 없다
}


sayHello()

// 리턴값이 있는 함수
@discardableResult
func sayHelloString() -> String {
    print("하이")
    return "안녕하세요"  //프린트동작도하고 "안녕하세요" 라는 값도 반환한다.
}

sayHelloString()     // 실제 프로젝트에서는 경고창 표시를해준다 왜 리턴결과값이있는데 왜 사용을 안해??? 라고 한다. 그럴때는 사용해줘야한다
_ = sayHelloString() // 기존에는 이렇게 와이드 카드로도 사용했었다.



//리턴타입이 없는 함수는 결과값이 void 이고 cpu의 제어권만 가지고 있다.
//리턴 타입이 있는 함수는 결과값이 있고 cpu의 제어권과 결과값을 모두 가진다.

//아까 리턴값이 있는 함수에서 리턴값을 사용 하지않는다면 경고창을 표시해 준다고 했는데 그걸 기존에는 와일드카드를 이용해서 처리했었는데 스위프트 5.2 이후부터는 이렇게 처리한다.

// 기존의 방식

_ = sayHelloString()    // 지금까지 많이 사용해오던 패턴



// 스위프트 5.2 부터

@discardableResult   //어트리뷰트를 사용하도록함 이걸 함수위에 써주어서 굳이 사용안해도 버릴수도 있는 결과야 라고 컴파일러한테 알려준다.
```
