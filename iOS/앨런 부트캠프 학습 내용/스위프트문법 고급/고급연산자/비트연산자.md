```swift
//비트 연산자 (앱개발에는 잘 사용이안됨 가볍게 보자) =====================

/*
 비트연산자란 메모리 비트(bit)단위로 직접적인 논리연산을 하거나 , 비트 단위 이동시에 사용하는 연산자
 하드웨어적인처리, 그래픽프로그래밍, 임베디드 프로그래밍,게임프로그래밍 등 아주 한정적으로 쓰인다
 장접: 연산속도가 빠르다, 잛은코드로 복잡한 로직을 구현가느안 경우가 있다.
 
 스위프트에는 6가지가 있다
 
 논리연산
 ~ 비트와이즈 낫 연산자
 & 비트와이즈 앤드 연산자
 | 비트와이즈 오어 연산자
 ^ 비트와이즈 엑스오어 연산자
 
 이동연산
 << 비트와이즈 레프트 시프트 연산자
 >> 비트와이즈 라이트 시프트 연산자
 */


//비트와이즈 낫 연산자 ======================

//비트 논리 부정연산자라고도 함, 단항 연산자의 형태로 사용
//기존의 메모르의 비트를 반전시킴 뒤집어버림
// 0은 1로 1은 0으로

let a1: UInt = 0b0000_1111
let b1 = ~a1 //0b1111_0000


//비트와이즈 앤드 연산자 =====================
//비트 논리 곱연산자라고도 함, 이항 연산자의 형태로 사용
//두개의 메모리의 비트중 모두가 1이면 1을 반환
//두개의 메모리의 비트중 둘다 1일때만 1로 설정 true && true 생각하면됨

let a2: UInt8 = 0b1111_1100
let b2: UInt8 = 0b0011_1111

a2 & b2 // 0b0011_1100
//걍 쉽게말해 겹치는 부분은 1 안겹치는부분은 0

//비트와이즈 오어 연산자 ========================
//비트 논리 합 연산자라고도 함 이항 연산자
//두개의 메모리의 비트중 하나라도 1이면 1을 반환 true || true 생각하면됨

let a3: UInt8 = 0b1111_1100
let b3: UInt8 = 0b0011_1110

a3 & b3 // 0b1111_1110

//비트와외자 엑스오어 연산자 ====================
//비트논리배타연산자 이항연산자임
//두개의 메모리의 비트중 둘을 비교해서 서로 다르면 1을 반환
//서로 다르면 ===> 1을반환, 서로같으면 ===> 0 을반환
//내가1이면 너는 0이여야 난 1을 만든꺼야~ 베타적임

let a4: UInt8 = 0b0001_0100
let b4: UInt8 = 0b0000_0101

a4 ^ b4 //0b0001_0001



//비트 이동연산자 ====================
//기존 비트를 요청된 값만큼 왼쪽이나 오른쪽으로 이동
//정수의 수용범위를 넘어서는 비트는 어떤것이든 버림
//비트를 왼쪽이나 오른쪽으로 이동하면서 남는 공간에는 0을 삽입
//대신 부호가 있는 경우에는 1 삽입 (원래의 부호를 유지하기 위한것) (라이트 일때만 조심)


let leftbits: UInt8 = 4 //0000_0100
leftbits << 1  //0000_1000
leftbits << 2  //0001_0000
leftbits << 5  //1000_0000

// 당연히 >>는 반대로


//라이트 시프트 연산자는 이걸 고려해야한다
//비트의 맨 앞에 부호의 정보가 담겨있는 경우 오른쪽으로 이동하는경우에
//부호는 유지되어야하니 1을 삽입한다.
```
